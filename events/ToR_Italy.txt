# Civil War
country_event = {
    id = ita.8
    title = ita.8.t
    desc = ita.8.d
    picture = GFX_report_event_french_resistance_01

    is_triggered_only = yes

    immediate = {
            set_global_flag = icw
		

            SRI = {

                if = {
                    limit = {
                        FROM = { controls_state = 158 }
                    }
                    transfer_state = 158
                }
                if = {
                    limit = {
                        FROM = { controls_state = 161 }
                    }
                    transfer_state = 161
                }
                if = {
                    limit = {
                        FROM = { controls_state = 162 }
                    }
                    transfer_state = 162
                }
                if = {
                    limit = {
                        FROM = { controls_state = 159 }
                    }
                    transfer_state = 159
                }
				
				every_state = {
					limit = {
						is_core_of = ITA
					}
					THIS = {
						add_core_of = PIE
					}
				}

                add_equipment_to_stockpile = {
                    type = infantry_equipment
                    amount = 10000
                    producer = SRI
                }
                add_manpower = 50000
                load_oob = SRI_rebellion
            }
			ITA = {
			add_popularity = {
				ideology = vanguardist
				popularity = -0.05
			}
			add_popularity = {
				ideology = orthodox_marxist
				popularity = -0.05
			}	
			add_popularity = {
				ideology = revolutionary_socialist
				popularity = -0.05
			}
		}
    }

    option = {
		name = ita.8.a
		ai_chance = {
			factor = 100
		}

        add_stability = 0.15
		add_war_support = 0.2

		SRI = {
			declare_war_on = {
                target = ITA
                type = annex_everything
            }
		}

        news_event = { hours = 6 id = tornews.3 }
    }
     
    option = {
		name = ita.8.b
		ai_chance = {
			factor = 0
		}
        add_stability = 0.15
		add_war_support = 0.2

		SRI = {
            change_tag_from = ITA
                }
		SRI = {
			declare_war_on = {
                target = ITA
                type = annex_everything
            }
		}

        news_event = { hours = 6 id = tornews.3 }
    }
}
