    id = ita.7

    desc = ita.7.d
    picture = GFX_report_event_french_resistance_01

    is_triggered_only = yes

    immediate = {
        hidden_effect = {
            set_global_flag = icw

            SRI = {
                set_country_flag = choice_made

                if = {
                    limit = {
                        FROM = { controls_state = 158 }
                    }
                    transfer_state = 158
                }
                if = {
                    limit = {
                        FROM = { controls_state = 161 }
                    }
                    transfer_state = 161
                }
                if = {
                    limit = {
                        FROM = { controls_state = 162 }
                    }
                    transfer_state = 162
                }
                if = {
                    limit = {
                        FROM = { controls_state = 159 }
                    }
                    transfer_state = 159
                }

                add_equipment_to_stockpile = {
                    type = infantry_equipment
                    amount = 1000
                    producer = SRI
                }
                add_manpower = 30000
                load_oob = SRI_rebellion
            }
        }
    }

    option = {
		name = ita.7.a
		ai_chance = {
			factor = 100
		}

        add_stability = 0.15
		add_war_support = 0.2

		SRI = {
			delare_war_on = {
                target = ITA
                type = annex_everything
            }
		}

        news_event = { hours = 6 id = tornews.3 }
    }
     
    option = {
		name = ita.7.b
		ai_chance = {
			factor = 0
		}
        add_stability = 0.15
		add_war_support = 0.2

		SRI = {
            change_tag_from = ITA
			delare_war_on = {
                target = ITA
                type = annex_everything
            }
		}

        news_event = { hours = 6 id = tornews.3 }
    }
}
