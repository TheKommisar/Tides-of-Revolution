ICH_subversive_operations = {
    ICH_prepare_coup = {
        target_trigger = {
            FROM = {
                OR = {
                    tag = SIA
                    tag = PHI
                    tag = RAJ
                    tag = INS
                }
            }
        }

        visible = {
            FROM = { has_country_flag = ICH_COUP_ALLOWED_FLAG }
        }

        cost = 50
        days_remove = 60
        fire_only_once = yes
        
        complete_effect = {
            set_country_flag = INS_PREPARING_COUP_FLAG
            FROM = { set_country_flag = ICH_ONGOING_TARGET_FLAG }
        }

        remove_effect = {
            clr_country_flag = INS_PREPARING_COUP_FLAG
        }
    }

    ICH_ignite_coup = {
        target_trigger = {
            FROM = {
                OR = {
                    tag = SIA
                    tag = PHI
                    tag = RAJ
                    tag = INS
                }
            }
        }

        visible = {
            FROM = { has_country_flag = ICH_ONGOING_TARGET_FLAG }
        }

        cost = 150
        days_remove = 120
        fire_only_once = yes
        
        complete_effect = {
            FROM = {
                country_event = { id = ich.11 days = 1 }
            }
        }

        remove_effect = {
            clr_country_flag = INS_ONGOING_TARGET
        }
    }
}
